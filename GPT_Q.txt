$(document).ready(function () {
  $("#btn").click(function () {
    $.ajax({
      dataType: "json",
      url: "http://localhost:8080/training/res/S010010001",
      method: "POST",
      data: JSON.stringify({id: "A123456789"}),
      success: function (result) {
        // 回應的資料
        $(".data").text("成功");

        console.log("成功:", result);
      },
      error: function (error) {
        // 錯誤
        $(".data").text("失敗");
        console.log("失敗:", error);
      },
    });
  });
});
